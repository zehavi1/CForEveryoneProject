<expr> ::= <term> (TOK_PLUS <term> | TOK_MINUS <term>)*
<term> ::= <factor> (TOK_ASTERISK <factor> | TOK_SLASH <factor> | TOK_PERCENT <factor>)*
<factor> ::= TOK_ID | <number>  | TOK_OPEN_PAREN <expr> TOK_CLOSE_PAREN | TOK_INCREMENT TOK_ID | TOK_DECREMENT TOK_ID | TOK_ID TOK_INCREMENT | TOK_ID TOK_DECREMENT
<number> ::= TOK_INT | TOK_DOUBLE | TOK_FLOAT | TOK_CHAR

<condition> ::= <expr> (TOK_EQUAL | TOK_NOT_EQUAL | TOK_LESS | TOK_GREATER | TOK_LESS_EQUAL | TOK_GREATER_EQUAL) <expr>
<range_condition> ::= <expr> TOK_LESS <expr> TOK_LESS <expr> | <expr> TOK_GREATER <expr> TOK_GREATER <expr>

<if_statement> ::= TOK_IF TOK_OPEN_PAREN <condition> TOK_CLOSE_PAREN <block>
<if_range_statement> ::= TOK_IFRANGE TOK_OPEN_PAREN <range_condition> TOK_CLOSE_PAREN <block>
<if_else_statement> ::= TOK_IF TOK_OPEN_PAREN <condition> TOK_CLOSE_PAREN <block> TOK_ELSE <block>
<elif_statement> ::= TOK_IF TOK_OPEN_PAREN <condition> TOK_CLOSE_PAREN <block> (TOK_ELIF TOK_OPEN_PAREN <condition> TOK_CLOSE_PAREN <block>)* (TOK_ELSE <block>)?

<loop> ::= <for_loop> | <while_loop> | <foreach_loop>
<for_loop> ::= TOK_FOR TOK_OPEN_PAREN <assignment> TOK_SEMICOLON <condition> TOK_SEMICOLON <assignment> TOK_CLOSE_PAREN <block>
<while_loop> ::= TOK_WHILE TOK_OPEN_PAREN <condition> TOK_CLOSE_PAREN <block>
<foreach_loop> ::= TOK_FOREACH TOK_OPEN_PAREN TOK_ID TOK_IN <collection> TOK_CLOSE_PAREN <block>
<collection> ::= TOK_ID | TOK_LEFT_ARRAY (<expr> (TOK_COMMA <expr>)*)? TOK_RIGHT_ARRAY

<block> ::= TOK_OPEN_CURLY <statement>* TOK_CLOSE_CURLY
<statement> ::= <if_statement> 
              | <if_range_statement> 
              | <elif_statement> 
              | <loop> 
              | <assignment> 
              | <print_statement> 
              | TOK_RETURN <expr> TOK_SEMICOLON

<assignment> ::= (<type>|TOK_VAR) TOK_ID ((TOK_PLUS | TOK_MINUS | TOK_ASTERISK | TOK_SLASH)  TOK_ASSIGN <expr>) | (TOK_INCREMENT | TOK_DECREMENT) (TOK_ASSIGN <expr>)? TOK_SEMICOLON
<type> ::=TOK_DOUBLE_TYPE | TOK_INT_TYPE | TOK_CHAR_TYPE |
                 TOK_BOOL_TYPE | TOK_FLOAT_TYPE | TOK_STRING_TYPE | TOK_LONG_TYPE
<print_statement> ::= TOK_PRINT TOK_OPEN_PAREN <expr> TOK_CLOSE_PAREN TOK_SEMICOLON

<parameter_list> ::= (<type> TOK_ID (TOK_COMMA <type> TOK_ID)*)?
<function_definition> ::= (<type>|TOK_VOID) TOK_ID TOK_OPEN_PAREN <parameter_list> TOK_CLOSE_PAREN <block>
